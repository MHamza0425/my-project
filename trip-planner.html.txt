<script>
  // Dark mode toggle
  document.getElementById("darkToggle").addEventListener("click", () => {
    document.body.classList.toggle("dark");
  });

  // Mobile menu toggle
  document.getElementById("mobileMenuBtn").addEventListener("click", () => {
    document.getElementById("mobileMenu").classList.toggle("hidden");
  });

  // Trip Planner logic
  const addTripBtn = document.getElementById("addTripBtn");
  const tripList = document.getElementById("tripList");

  addTripBtn.addEventListener("click", () => {
    const destination = document.getElementById("destination").value.trim();
    const date = document.getElementById("tripDate").value;
    const notes = document.getElementById("notes").value.trim();

    if (!destination || !date) {
      alert("Please enter a destination and date.");
      return;
    }

    const li = document.createElement("li");
    li.className = "bg-gray-100 dark:bg-gray-700 rounded p-3 shadow flex justify-between items-center";
    li.innerHTML = `
      <div>
        <strong>${destination}</strong> - ${new Date(date).toLocaleDateString()}<br>
        ${notes ? `<em>${notes}</em>` : ''}
      </div>
      <button class="text-red-600 dark:text-red-400 hover:underline" onclick="this.parentElement.remove()">Delete</button>
    `;

    tripList.appendChild(li);

    // Clear inputs
    document.getElementById("destination").value = '';
    document.getElementById("tripDate").value = '';
    document.getElementById("notes").value = '';
  });
</script>
